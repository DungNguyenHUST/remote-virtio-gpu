add_library(rvgpu SHARED
	tcp/rvgpu-tcp.c
	zmq/rvgpu-zmq.c
	res/rvgpu-res.c
	rvgpu.c
)
set_target_properties(rvgpu
	PROPERTIES
	VERSION 1.0.0
	SOVERSION 1
	)
target_link_libraries(rvgpu PRIVATE pthread zmq pgm)
target_include_directories(rvgpu
	PRIVATE
		${PROJECT_SOURCE_DIR}/include
		${extlibs_INCLUDE_DIRS}
	)
target_compile_definitions(rvgpu PRIVATE
	_DEFAULT_SOURCE
	_GNU_SOURCE)
install(TARGETS rvgpu
	ARCHIVE DESTINATION lib
	LIBRARY DESTINATION lib
	RUNTIME DESTINATION bin)
